-- Création de la base de données
CREATE DATABASE IF NOT EXISTS GestionStagiaires;
USE GestionStagiaires;

-- Suppression des tables si elles existent déjà pour éviter les erreurs
DROP TABLE IF EXISTS Controle;
DROP TABLE IF EXISTS Module;
DROP TABLE IF EXISTS Stagiaire;

-- Création de la table Stagiaire
CREATE TABLE Stagiaire (
    numIns VARCHAR(10) PRIMARY KEY,
    nom VARCHAR(50) NOT NULL,
    prenom VARCHAR(50) NOT NULL,
    DateN DATE,
    Groupe VARCHAR(20),
    Tel VARCHAR(15)
);

-- Création de la table Module
CREATE TABLE Module (
    idModule VARCHAR(10) PRIMARY KEY,
    intitule VARCHAR(100) NOT NULL,
    coeff INT
);

-- Création de la table Controle
CREATE TABLE Controle (
    numIns VARCHAR(10),
    idModule VARCHAR(10),
    date DATE,
    note DECIMAL(4,2),
    PRIMARY KEY (numIns, idModule, date),
    FOREIGN KEY (numIns) REFERENCES Stagiaire(numIns),
    FOREIGN KEY (idModule) REFERENCES Module(idModule)
);
